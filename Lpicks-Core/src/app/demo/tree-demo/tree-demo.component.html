  • How to listen to selecte item from drop down?

    • Use this EventListener in your selector tag to listen to the selected path and you have to create a event handler function in your component.

    Copy To Clipboard 
    (sendItem)="handler($event)" 
(sendLink)="handlerLink($event)"

handler(item: TreeItem) {
  this.item = item;
}

handlerLink(link: string) {
  this.link = link;
}
